# Git and GitHub Module Cheatsheet

Initialize:
    git init

Status:
    git status

Stage:
    git add .

Commit:
    git commit -m "Message"

History:
    git log --oneline

Branch:
    git switch -c branch-name

Merge:
    git merge branch-name

Push:
    git push

Pull:
    git pull

Clone:
    git clone URL
